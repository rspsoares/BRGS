CREATE PROCEDURE [dbo].[SP_GERADORES_USOS_GRID]
(	
	@IdGerador VARCHAR(10)
)
AS
BEGIN
	SELECT 
		G.IdGerador,
		G.DataInicio,
		G.DataFim,
		C.Nome AS 'NomeCliente',
		O.NumeroLicitacao
	FROM Geradores_Usos (NOLOCK) G
		INNER JOIN Clientes (NOLOCK) C On G.IdCliente = C.idCliente
		INNER JOIN ObrasEtapas(NOLOCK) O ON G.IdEtapaObra = O.idObraEtapa
	WHERE		
		G.IdGerador = @IdGerador
END

ALTER PROCEDURE [dbo].[SP_GERADORES_CONSULTAR]
(
	@Id INT = NULL,
	@IdEmpresa INT = NULL,
	@IdCliente VARCHAR(10) = NULL,
	@IdObraEtapa VARCHAR(10) = NULL,
    @NumeroSerie VARCHAR(30) = NULL, 
	@Combustivel VARCHAR(30) = NULL,
    @Marca VARCHAR(30) = NULL,
	@Modelo VARCHAR(30) = NULL,    
	@Lotado VARCHAR(50) = NULL,
    @NotaFiscal VARCHAR(30) = NULL,
    @DataCompra DATETIME = NULL,
    @Acessorios VARCHAR(MAX) = NULL
)
AS
BEGIN
	SELECT ID
      ,G.IdEmpresa
      ,ISNULL(G.IdCliente,0) AS IdCliente
	  ,ISNULL(G.IdObraEtapa,0) AS IdObraEtapa	  
	  ,ISNULL(G.DataAlocacao, CONVERT(DATETIME, '1753-01-01', 102)) AS DataAlocacao
      ,G.NumeroSerie   
	  ,G.Combustivel
	  ,G.Marca
      ,G.Modelo      
	  ,G.Lotado
      ,G.NotaFiscal
      ,G.DataCompra
      ,G.Acessorios
	  ,C.Nome AS 'Cliente'
	FROM dbo.Geradores (NOLOCK) G
		LEFT JOIN Clientes (NOLOCK) C ON G.IdCliente = C.idCliente
	WHERE
		(G.ID = @Id OR @ID IS NULL) AND
		(G.IdEmpresa = @IdEmpresa OR @IdEmpresa IS NULL) AND
		(G.IdCliente = @IdCliente OR @IdCliente IS NULL) AND
		(G.IdObraEtapa = @IdObraEtapa OR @IdObraEtapa IS NULL) AND
		(G.NumeroSerie LIKE '%' + @NumeroSerie + '%' OR @NumeroSerie IS NULL) AND		
		(G.Modelo LIKE '%' + @Modelo + '%' OR @Modelo IS NULL) AND		
		(G.Lotado LIKE '%' + @Lotado + '%' OR @Lotado IS NULL) AND		
		(G.NotaFiscal = @NotaFiscal or @NotaFiscal IS NULL) AND
		(G.DataCompra = @DataCompra or @DataCompra IS NULL) AND
		(G.Acessorios LIKE '%' + @Acessorios + '%' OR @Acessorios IS NULL)
END

CREATE PROCEDURE [dbo].[SP_GERADORES_LIBERAR]
	@Id VARCHAR(10),
	@idCliente VARCHAR(10),
	@idObraEtapa VARCHAR(10)	
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION
			DECLARE @DataAlocacao DATETIME

			SELECT @DataAlocacao = DataAlocacao FROM Geradores WHERE Id = @Id

			UPDATE Geradores
			SET
				IdCliente = NULL,
				IdObraEtapa = NULL,
				DataAlocacao = NULL
			WHERE
				 Id = @Id

			INSERT INTO Geradores_Usos
				   ([IdGerador]
				   ,[IdCliente]
				   ,[IdEtapaObra]
				   ,[DataInicio]
				   ,[DataFim])
			 VALUES
				   (@Id
				   ,@idCliente
				   ,@idObraEtapa 
				   ,@DataAlocacao
				   ,GETDATE())

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		DECLARE @ErrorNumber INT = ERROR_NUMBER();
		DECLARE @ErrorLine INT = ERROR_LINE();
		DECLARE @ErrorMessage NVARCHAR(4000) = ERROR_MESSAGE();
		DECLARE @ErrorSeverity INT = ERROR_SEVERITY();
		DECLARE @ErrorState INT = ERROR_STATE();
		
		IF @@TRANCOUNT > 0
		BEGIN
			ROLLBACK TRANSACTION
		END;

		RAISERROR(@ErrorMessage, @ErrorSeverity, @ErrorState);		
		
	END CATCH;
END
